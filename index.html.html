def F_der(x):
  return (x > -1).astype(int)

def Psi_der(x):
  return (-(x < 1).astype(int))

#soft_max tonos
def f2_der(x):
  return np.exp(-x)/np.power((1+np.exp(-x)), 2)

#h paragwgos ths ReLu poy einai 1 gia x>0, kai 0 gia x<=0
def f1_der(x):
  return (x > 0).astype(int)

def Relu(x):
  return np.maximum(0,x)

def cost(x,y):
  return (np.maximum(0,1+x) + np.maximum(0,1-y))

def Soft(x):
  return 1/ (1 + np.exp(-x))

def prop_1(c):
  #c = a[:,i]
  W_1 = np.matmul(A_1,c) + B_1
  Z_1 = Relu(W_1)
  W_2 = np.matmul(A_2, Z_1) + B_2
  Y = W_2
  #Y = Soft(W_2)
  return Y, W_2, Z_1, W_1

def prop_2(d):
  W_1_a = np.matmul(A_1,d) + B_1
  Z_1_a = Relu(W_1_a)
  W_2_a = np.matmul(A_2, Z_1_a) + B_2
  Y_a = W_2_a
  #Y_a = Soft(W_2_a)
  return Y_a, W_2_a, Z_1_a, W_1_a

def update_par(c, d, A_1, B_1, A_2, B_2):
  m=0.0001

  Y, W_2, Z_1, W_1 = prop_1(c)
  Y_a, W_2_a, Z_1_a, W_1_a = prop_2(d)

  B_2 = B_2 - m*(F_der(Y)*f2_der(W_2) + Psi_der(Y_a)*f2_der(W_2)) #prepei na allaksw deuteros oros, ws pros x1

  A_2 = A_2 - m*(F_der(Y)*f2_der(W_2)*Z_1 + Psi_der(Y_a)*f2_der(W_2)*Z_1_a) #prepei na allaksw deuteros oros, ws pros x1

  B_1 = B_1 - m*(np.multiply(F_der(Y)*f2_der(W_2)*A_2, f1_der(W_1)) + np.multiply(Psi_der(Y_a)*f2_der(W_2_a)*A_2, f1_der(W_1_a)))  #prepei na allaksw deuteros oros, ws pros x1

  A_1 = A_1 - m*(np.transpose(np.matmul(c.reshape(784,1),np.multiply(F_der(Y)*f2_der(W_2)*A_2, f1_der(W_1)).reshape(1,300))) + np.transpose(np.matmul(d.reshape(784,1),np.multiply(Psi_der(Y_a)*f2_der(W_2_a)*A_2, f1_der(W_1_a)).reshape(1,300))))

  kostos = cost(Y, Y_a)
  return A_1, B_1, A_2, B_2, kostos
